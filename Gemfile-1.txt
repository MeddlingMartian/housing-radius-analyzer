# Gemfile for housing-radius-analyzer Jekyll site
# Requires Ruby >= 3.1  (setup-ruby@v1 will install this automatically)

source "https://rubygems.org"

# Jekyll core
gem "jekyll", "~> 4.3"

# GitHub Pages compatible theme
gem "minima", "~> 2.5"

# Jekyll plugins (safe for GitHub Pages)
group :jekyll_plugins do
  gem "jekyll-feed",     "~> 0.12"
  gem "jekyll-seo-tag",  "~> 2.8"
  gem "jekyll-sitemap"
  gem "jekyll-relative-links"
end

# Windows / JRuby compatibility shims
platforms :mingw, :x64_mingw, :mswin, :jruby do
  gem "tzinfo",      ">= 1", "< 3"
  gem "tzinfo-data"
end

# Performance: avoid regenerating files that haven't changed
gem "wdm", "~> 0.1.1", :platforms => [:mingw, :x64_mingw, :mswin]

# Lock Webrick for Ruby 3+
gem "webrick", "~> 1.8"
